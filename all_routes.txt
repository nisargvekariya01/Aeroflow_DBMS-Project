    const { searchParams } = new URL(request.url);
    const id = searchParams.get('id');
    if (!id) return NextResponse.json({ error: 'ID is required' }, { status: 400 });

    // Programmatic Bottom-Up Cascading Deletion
    const flightRes = await query('SELECT Flight_ID FROM Flight WHERE Aircraft_ID = $1', [id]);
    if (flightRes.rows.length > 0) {
       const flightIds = flightRes.rows.map(r => r.flight_id);
       const qs = flightIds.map((_, i) => `$${i+1}`).join(',');
       
       await query(`DELETE FROM Luggage WHERE Booking_ID IN (SELECT Booking_ID FROM Booking WHERE Flight_ID IN (${qs}))`, flightIds);
       await query(`DELETE FROM Booking WHERE Flight_ID IN (${qs})`, flightIds);
       await query(`DELETE FROM Crew_Assign WHERE Flight_ID IN (${qs})`, flightIds);
       await query(`DELETE FROM Pilot_Assign WHERE Flight_ID IN (${qs})`, flightIds);
       await query(`DELETE FROM Uses_Gate WHERE Flight_ID IN (${qs})`, flightIds);
       await query(`DELETE FROM Uses_Runway WHERE Flight_ID IN (${qs})`, flightIds);
       await query(`DELETE FROM Flight_Legs WHERE Flight_ID IN (${qs})`, flightIds);
       await query(`DELETE FROM Flight WHERE Aircraft_ID = $1`, [id]);
    }
    
    await query(`DELETE FROM Maintenance WHERE Aircraft_ID = $1`, [id]);
    await query(`DELETE FROM Aircraft WHERE Aircraft_ID = $1`, [id]);

    return NextResponse.json({ message: 'Aircraft and all dependent records deleted successfully' });

import { query } from '@/lib/db';
import { NextResponse } from 'next/server';

export async function GET() {
  try {
    const result = await query(
      `SELECT f.*, s.IATA_Code AS source_iata, d.IATA_Code AS dest_iata 
       FROM Flight f 
       JOIN Airport s ON f.Source_Airport_ID = s.Airport_ID 
       JOIN Airport d ON f.Dest_Airport_ID = d.Airport_ID 
       ORDER BY f.Flight_ID`
    );
    return NextResponse.json(result.rows);
  } catch (error) {
    return NextResponse.json({ error: error.message }, { status: 500 });
  }
}

export async function POST(request) {
  try {
    const { id, aircraft_id, departure_time, arrival_time, source_airport_id, dest_airport_id } = await request.json();

    if (!id || !aircraft_id || !departure_time || !arrival_time || !source_airport_id || !dest_airport_id) {
      return NextResponse.json({ error: 'All flight fields are required.' }, { status: 400 });
    }

    // Check Aircraft availability constraints
    const ac = await query('SELECT Status_Type FROM Aircraft WHERE Aircraft_ID = $1', [aircraft_id]);
    if (ac.rows.length === 0) {
      return NextResponse.json({ error: `Aircraft with ID ${aircraft_id} not found.` }, { status: 404 });
    }
    if (String(ac.rows[0].status_type).toUpperCase() !== 'AVAILABLE') {
      return NextResponse.json({ error: `Aircraft ${aircraft_id} is unavailable (${ac.rows[0].status_type}).` }, { status: 400 });
    }

    const existing = await query('SELECT Flight_ID FROM Flight WHERE Flight_ID = $1', [id]);
    if (existing.rows.length > 0) {
      return NextResponse.json({ error: `Flight with ID ${id} already exists.` }, { status: 409 });
    }

    await query(
      `INSERT INTO Flight (Flight_ID, Aircraft_ID, Departure_Time, Arrival_Time, Source_Airport_ID, Dest_Airport_ID)
       VALUES ($1,$2,$3,$4,$5,$6)`,
      [id, aircraft_id, departure_time, arrival_time, source_airport_id, dest_airport_id]
    );

    // Calculate duration in hours, rounded up (Math.ceil)
    const durationMs = new Date(arrival_time).getTime() - new Date(departure_time).getTime();
    const additionalHours = Math.ceil(durationMs / (1000 * 60 * 60));

    // Update Aircraft status to Active and increment hours
    await query(`UPDATE Aircraft SET Status_Type = 'ACTIVE', Total_Flight_Hours = COALESCE(Total_Flight_Hours, 0) + $2 WHERE Aircraft_ID = $1`, [aircraft_id, additionalHours]);

    return NextResponse.json({ message: 'Flight added successfully' });
  } catch (error) {
    return NextResponse.json({ error: error.message }, { status: 500 });
  }
}

export async function PUT(request) {
  try {
    const { id, aircraft_id, departure_time, arrival_time, source_airport_id, dest_airport_id } = await request.json();
    const oldFlight = await query('SELECT Aircraft_ID, Departure_Time, Arrival_Time FROM Flight WHERE Flight_ID = $1', [id]);
    const oldFlightData = oldFlight.rows.length > 0 ? oldFlight.rows[0] : null;
    const oldAircraftId = oldFlightData?.aircraft_id;

    const newDurationMs = new Date(arrival_time).getTime() - new Date(departure_time).getTime();
    const newHours = Math.ceil(newDurationMs / (1000 * 60 * 60));

    if (String(oldAircraftId) !== String(aircraft_id)) {
      // Check new aircraft availability
      const ac = await query('SELECT Status_Type FROM Aircraft WHERE Aircraft_ID = $1', [aircraft_id]);
      if (ac.rows.length === 0) {
        return NextResponse.json({ error: `Aircraft with ID ${aircraft_id} not found.` }, { status: 404 });
      }
      if (String(ac.rows[0].status_type).toUpperCase() !== 'AVAILABLE') {
        return NextResponse.json({ error: `Aircraft ${aircraft_id} is unavailable (${ac.rows[0].status_type}).` }, { status: 400 });
      }
      
      // Safe to release the old aircraft and rollback its hours
      if (oldAircraftId) {
        const oldDurationMs = (oldFlightData.departure_time && oldFlightData.arrival_time) ? new Date(oldFlightData.arrival_time).getTime() - new Date(oldFlightData.departure_time).getTime() : 0;
        const oldHours = Math.ceil(oldDurationMs / (1000 * 60 * 60));
        await query(`UPDATE Aircraft SET Status_Type = 'AVAILABLE', Total_Flight_Hours = GREATEST(0, COALESCE(Total_Flight_Hours, 0) - $2) WHERE Aircraft_ID = $1`, [oldAircraftId, oldHours]);
      }
    }

    await query(
      `UPDATE Flight SET Aircraft_ID=$2, Departure_Time=$3, Arrival_Time=$4, Source_Airport_ID=$5, Dest_Airport_ID=$6
       WHERE Flight_ID = $1`,
      [id, aircraft_id, departure_time, arrival_time, source_airport_id, dest_airport_id]
    );

    if (String(oldAircraftId) === String(aircraft_id)) {
        // Delta adjustment for the same plane
